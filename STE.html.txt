  <script>
    document.getElementById('year').textContent = new Date().getFullYear();

    function handleForm(e){
      e.preventDefault();
      const name = document.getElementById('name').value.trim();
      const phone = document.getElementById('phone').value.trim();
      const message = document.getElementById('message').value.trim();
      const status = document.getElementById('formStatus');
      // Basic fallback: open mail client
      const mailto = `mailto:info@sagortelecom.com?subject=${encodeURIComponent('Contact from '+name)}&body=${encodeURIComponent('Phone: '+phone+'\n\n'+message)}`;
      status.textContent = 'আপনার ডিফল্ট ইমেইল অ্যাপে তথ্য খুলছে...';
      window.location.href = mailto;
    }

    // Small helper: swap logo if user drags a file onto logo (local preview only)
    const logo = document.getElementById('logo');
    logo.addEventListener('dragover', e=>e.preventDefault());
    logo.addEventListener('drop', e=>{
      e.preventDefault();
      const f = e.dataTransfer.files[0];
      if(!f) return;
      const reader = new FileReader();
      reader.onload = ev => logo.src = ev.target.result;
      reader.readAsDataURL(f);
      alert('লোকাললি লোগো প্রিভিউ করা হচ্ছে — ডাটাটি ডাটাবেসে আপলোড হবে না।');
    });
  </script>
